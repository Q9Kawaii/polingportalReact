# 1. Initialize a new Git repository
git init  

# 2. Add all files to the staging area
git add .  

# 3. Commit the changes with a message
git commit -m "Initial commit"  

# 4. Set the default branch to 'main' (optional but recommended)
git branch -M main  

# 5. Add the remote repository URL
git remote add origin <your-repo-url>  

# 6. Push the initial commit to the remote repository
git push -u origin main  

if some shit happens, [ git pull origin main --rebase ] type this then [ git push -u origin main ]

function add_git_hehe {
